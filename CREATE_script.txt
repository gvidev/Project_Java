CREATE TABLE STUDENTS (
STUDENT_ID IDENTITY(1,1) PRIMARY KEY,
NAME NVARCHAR(40) NOT NULL,
EMAIL NVARCHAR(50) NOT NULL,
PHONE_NUMBER NVARCHAR(10) NOT NULL,
GENDER NVARCHAR(10) NOT NULL,
MAJOR NVARCHAR(50) NOT NULL,
FACULTY_NUMBER NVARCHAR(10) NOT NULL
);

CREATE TABLE COURSES (
  COURSE_ID IDENTITY(1,1) PRIMARY KEY,
  COURSE_NAME NVARCHAR(20) NOT NULL,
  TEACHER_NAME NVARCHAR(50) NOT NULL,
  CREDIT_HOURS INT NOT NULL,
  DESCRIPTION NVARCHAR(100) NOT NULL
);

CREATE TABLE ASSESSMENTS (
  ASSESSMENT_ID IDENTITY(1,1) PRIMARY KEY,
  STUDENT_ID INT NOT NULL,
  COURSE_ID INT NOT NULL,
  FINAL_GRADE INT NOT NULL,
  NOTES NVARCHAR(20) NOT NULL,
  ASSESSMENT_DATE DATE NOT NULL,
  FOREIGN KEY (STUDENT_ID) REFERENCES STUDENTS(STUDENT_ID),
  FOREIGN KEY (COURSE_ID) REFERENCES COURSES(COURSE_ID)
);
